import java.util.Random;
import java.util.Scanner;

public class LosowaTablica {
    public static void main(String[] args) {
        // Inicjalizacja skanera do pobrania danych od użytkownika
        Scanner scanner = new Scanner(System.in);

        // Pobieranie wartości x i y
        System.out.print("Podaj wartość x (początek zakresu): ");
        int x = scanner.nextInt();
        
        System.out.print("Podaj wartość y (koniec zakresu): ");
        int y = scanner.nextInt();

        // Sprawdzanie, czy x jest mniejsze od y
        if (x >= y) {
            System.out.println("Błąd: x musi być mniejsze niż y.");
            return;
        }

        // Tworzymy tablicę o losowej długości (możesz ustawić na stałą wartość, jeśli chcesz)
        System.out.print("Podaj rozmiar tablicy: ");
        int rozmiar = scanner.nextInt();
        
        // Tworzymy tablicę
        int[] tablica = new int[rozmiar];
        
        // Generator liczb losowych
        Random random = new Random();

        // Wypełniamy tablicę losowymi liczbami w zakresie od x do y
        for (int i = 0; i < tablica.length; i++) {
            tablica[i] = random.nextInt(y - x + 1) + x; // Losowa liczba w przedziale [x, y]
        }

        // Wyświetlanie wyników
        System.out.println("Tablica z losowymi liczbami w przedziale [" + x + ", " + y + "]:");
        for (int liczba : tablica) {
            System.out.print(liczba + " ");
        }
    }
}
